NET_DEV_SMART_PARAM_AREA

设备区域配置参数。

    OS_UINT8                       bySmartAreaVisible;
    OS_UINT8                       bySmartDirection;

    OS_UINT16                      wSmartSensitivity;
    OS_UINT16                      wSmartTime;
    OS_UINT16                      wSmartSizeRate;
    OS_UINT32                      byReserved[20];
} NET_DEV_SMART_PARAM_AREA;

NET_DEV_DISK_FILE

单个录像文件信息。

typedef struct {
    OS_UINT32        dwRecTrigMode;
    OS_UINT32        dwRecFileSize;
    char             sRecFileName[128];
    OS_UINT32        dwRecFileTime[2];
    char             sRecBeginTime[32];
    char             sRecEndTime[32];
    OS_UINT8         byReserved[128];
} NET_DEV_DISK_FILE;

dwRecFileSize
文件大小，单位：Byte
sRecFileName
录像文件名
dwRecFileTime
录像文件起始和结束时间，dwRecFileTime[0]-起始时间 dwRecFileTime[1]-结束时间，单位：秒
sRecBeginTime
录像文件起始时间，格式为“YYYY-MM-DD HH:MM:SS”
sRecEndTime
录像文件结束时间，格式为“YYYY-MM-DD HH:MM:SS”
byReserved
保留位，置为0

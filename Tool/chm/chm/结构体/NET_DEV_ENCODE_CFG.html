NET_DEV_ENCODE_CFG

码流编码信息。

    OS_UINT32              dwStreamNum;
    NET_DEV_ENCODE_INFO    struEncodeInfo[NET_MAX_STREAM];
    OS_UINT8               byRes[400];
} NET_DEV_ENCODE_CFG;
